import os
import smtplib
from email.mime.text import MIMEText
from dotenv import load_dotenv

load_dotenv()


def send_phishing_email(receiver_email, employee_id, campaign_id, sender="email1"):

    BASE_URL = os.getenv("BASE_URL")

    tracking_link = f"{BASE_URL}/track/{employee_id}/{campaign_id}"

    if sender == "email1":
        sender_email = os.getenv("EMAIL1_ADDRESS")
        sender_password = os.getenv("EMAIL1_PASSWORD")
    else:
        sender_email = os.getenv("EMAIL2_ADDRESS")
        sender_password = os.getenv("EMAIL2_PASSWORD")

    body = f"""
    <html>
    <body>

    <h2>Account Verification Required</h2>

    <p>Your organization account requires verification.</p>

    <a href="{tracking_link}">
        Verify Account
    </a>

    </body>
    </html>
    """

    msg = MIMEText(body, "html")
    msg["Subject"] = "Security Alert"
    msg["From"] = sender_email
    msg["To"] = receiver_email

    server = smtplib.SMTP("smtp.gmail.com", 587)
    server.starttls()
    server.login(sender_email, sender_password)

    server.sendmail(
        sender_email,
        receiver_email,
        msg.as_string()
    )

    server.quit()

This is actually how many enterprise systems work. They don't hardcode email credentials; they store multiple sender accounts (or use a centralized mail service) and associate actions with authenticated users.

What I would not do

I would not store both Gmail App Passwords in a shared .env file that's committed to Git. Even for an internship, that's not good security practice because anyone with access to the repository could use those Gmail accounts.

Instead:

Keep the database shared.
Give each teammate their own login to the application.
Store the sender account information securely (for a prototype, in a database with appropriate access controls; for production, in a secrets manager or encrypted storage).
Record which logged-in user sent each campaign.

That gives you one application, one database, both team members contributing, and a design that's much closer to what a real organization would build.
